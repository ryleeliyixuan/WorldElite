活动列表

method：POST
url: /api/activity/list

参数列表

{
  "page": "int //页数",
  "limit": "int //每页条数",
  "sort": "string //排序",
  "cityId": "int //所在城市",
  "cityIds": "int[] //城市ID",
  "status": "byte //活动状态",
  "title": "string //活动标题",
  "keyword": "string //搜索关键词"
}

返回结果: Android Code | IOS Code

{
  "code": "int //0为成功，其他错误码参考 ApiCode",
  "msg": "string //错误消息",
  "data": {
    "total": "int //总记录数",
    "pageCount": "int //页数",
    "currentPage": "int //当前页",
    "pageSize": "int //每页记录数",
    "list": [
      {
        "id": "int //活动ID",
        "title": "string //活动标题",
        "thumbnail": "string //缩略图",
        "summary": "string //摘要",
        "url": "string //链接",
        "description": "string //活动详情",
        "city": {
          "id": "int //字典ID",
          "name": "string //字典名",
          "type": "int //字典类型",
          "value": "string //字典值"
        },
        "startTime": "date //开始时间",
        "finishTime": "date //结束时间",
        "address": "string //活动地址",
        "status": "byte //状态",
        "joinFlag": "boolean //是否参加",
        "joinTime": "date //参加时间",
        "organizer": "string //主办方"
      }
    ],
    "hasMore": "boolean //是否还有更多"
  }
}
