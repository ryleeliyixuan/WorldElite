邮箱注册

method：POST
url: /api/auth/register

参数列表

参数名类型必需描述

返回结果: Android Code | IOS Code

{
  "code": "int //0为成功，其他错误码参考 ApiCode",
  "msg": "string //错误消息",
  "data": {
    "id": "long //用于后端",
    "userId": "string //字符串ID，用于前端",
    "name": "string //名字",
    "avatar": "string //头像",
    "gender": "byte //性别",
    "type": "byte //用户类型",
    "status": "byte //状态",
    "token": "string //登录凭证",
    "phoneCode": "string //电话区号",
    "phone": "string //电话号码",
    "email": "string //邮箱",
    "createTime": "date //创建时间",
    "subscribeFlag": "byte //订阅标志"
  }
}
