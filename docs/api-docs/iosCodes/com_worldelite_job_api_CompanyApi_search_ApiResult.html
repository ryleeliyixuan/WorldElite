#import <Foundation/Foundation>;

@interface ApiResult : NSObject

//0为成功，其他错误码参考 ApiCode
@property (nonatomic, strong) NSInteger code;
//错误消息
@property (nonatomic, assign) NSString msg;
//数据包
@property (nonatomic, strong) PageResult data;


@end



#import <Foundation/Foundation>;

@interface PageResult : NSObject

//总记录数
@property (nonatomic, strong) NSInteger total;
//页数
@property (nonatomic, strong) NSInteger pageCount;
//当前页
@property (nonatomic, strong) NSInteger currentPage;
//每页记录数
@property (nonatomic, strong) NSInteger pageSize;
//列表数据
@property (nonatomic, strong) CompanyVo[] list;
//是否还有更多
@property (nonatomic, strong) BOOL hasMore;


@end



#import <Foundation/Foundation>;

@interface CompanyVo : NSObject

//公司ID
@property (nonatomic, assign) NSString id;
//公司简称
@property (nonatomic, assign) NSString name;
//公司全称
@property (nonatomic, assign) NSString fullName;
//logo
@property (nonatomic, assign) NSString logo;
//主页
@property (nonatomic, assign) NSString homepage;
//公司介绍
@property (nonatomic, assign) NSString introduction;
//公司一句话介绍
@property (nonatomic, assign) NSString synopsis;
//公司规模
@property (nonatomic, strong) DictVo scale;
//公司阶段
@property (nonatomic, strong) DictVo stage;
//所属行业
@property (nonatomic, strong) DictVo industry;
//公司性质
@property (nonatomic, strong) DictVo property;
//公司地址
@property (nonatomic, strong) CompanyAddressVo[] addressList;
//完善进度
@property (nonatomic, strong) NSInteger completeProgress;
//百科摘要
@property (nonatomic, assign) NSString wikiSummary;
//公司百科
@property (nonatomic, assign) NSString companyWiki;
//收藏标志
@property (nonatomic, strong) int favoriteFlag;
//收藏时间
@property (nonatomic, assign) NSString favoriteTime;


@end
