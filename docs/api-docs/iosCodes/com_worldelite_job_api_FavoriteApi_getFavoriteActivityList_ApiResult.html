#import <Foundation/Foundation>;

@interface ApiResult : NSObject

//0为成功，其他错误码参考 ApiCode
@property (nonatomic, strong) NSInteger code;
//错误消息
@property (nonatomic, assign) NSString msg;
//数据包
@property (nonatomic, strong) PageResult data;


@end



#import <Foundation/Foundation>;

@interface PageResult : NSObject

//总记录数
@property (nonatomic, strong) NSInteger total;
//页数
@property (nonatomic, strong) NSInteger pageCount;
//当前页
@property (nonatomic, strong) NSInteger currentPage;
//每页记录数
@property (nonatomic, strong) NSInteger pageSize;
//列表数据
@property (nonatomic, strong) ActivityVo[] list;
//是否还有更多
@property (nonatomic, strong) BOOL hasMore;


@end



#import <Foundation/Foundation>;

@interface ActivityVo : NSObject

//活动ID
@property (nonatomic, strong) NSInteger id;
//活动标题
@property (nonatomic, assign) NSString title;
//缩略图
@property (nonatomic, assign) NSString thumbnail;
//摘要
@property (nonatomic, assign) NSString summary;
//链接
@property (nonatomic, assign) NSString url;
//活动详情
@property (nonatomic, assign) NSString description;
//活动城市
@property (nonatomic, strong) DictVo city;
//开始时间
@property (nonatomic, assign) NSString startTime;
//结束时间
@property (nonatomic, assign) NSString finishTime;
//活动地址
@property (nonatomic, assign) NSString address;
//状态
@property (nonatomic, strong) int status;
//是否参加
@property (nonatomic, strong) BOOL joinFlag;
//参加时间
@property (nonatomic, assign) NSString joinTime;


@end



#import <Foundation/Foundation>;

@interface DictVo : NSObject

//字典ID
@property (nonatomic, strong) NSInteger id;
//字典名
@property (nonatomic, assign) NSString name;
//字典类型
@property (nonatomic, strong) NSInteger type;
//字典值
@property (nonatomic, assign) NSString value;


@end
