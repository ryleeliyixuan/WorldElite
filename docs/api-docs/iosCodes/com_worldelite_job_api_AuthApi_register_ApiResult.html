#import <Foundation/Foundation>;

@interface ApiResult : NSObject

//0为成功，其他错误码参考 ApiCode
@property (nonatomic, strong) NSInteger code;
//错误消息
@property (nonatomic, assign) NSString msg;
//数据包
@property (nonatomic, strong) UserVo data;


@end



#import <Foundation/Foundation>;

@interface UserVo : NSObject

//用于后端
@property (nonatomic, strong) long id;
//字符串ID，用于前端
@property (nonatomic, assign) NSString userId;
//名字
@property (nonatomic, assign) NSString name;
//头像
@property (nonatomic, assign) NSString avatar;
//性别
@property (nonatomic, strong) int gender;
//用户类型
@property (nonatomic, strong) int type;
//状态
@property (nonatomic, strong) int status;
//登录凭证
@property (nonatomic, assign) NSString token;
//电话区号
@property (nonatomic, assign) NSString phoneCode;
//电话号码
@property (nonatomic, assign) NSString phone;
//邮箱
@property (nonatomic, assign) NSString email;
//创建时间
@property (nonatomic, assign) NSString createTime;
//订阅标志
@property (nonatomic, strong) int subscribeFlag;


@end
