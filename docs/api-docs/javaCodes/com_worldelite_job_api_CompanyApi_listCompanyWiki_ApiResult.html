public class ApiResult implements Serializable{

    private int code; //0为成功，其他错误码参考 ApiCode
    private String msg; //错误消息
    private PageResult data; //数据包

    @JSONField(name="code")
    public int getCode(){
        return code;
    }

    @JSONField(name="code")
    public void setCode(int code){
        this.code = code;
    }

    @JSONField(name="msg")
    public String getMsg(){
        return msg;
    }

    @JSONField(name="msg")
    public void setMsg(String msg){
        this.msg = msg;
    }

    @JSONField(name="data")
    public PageResult getData(){
        return data;
    }

    @JSONField(name="data")
    public void setData(PageResult data){
        this.data = data;
    }

}

public class PageResult implements Serializable{

    private int total; //总记录数
    private int pageCount; //页数
    private int currentPage; //当前页
    private int pageSize; //每页记录数
    private CompanyVo[] list; //列表数据
    private boolean hasMore; //是否还有更多

    @JSONField(name="total")
    public int getTotal(){
        return total;
    }

    @JSONField(name="total")
    public void setTotal(int total){
        this.total = total;
    }

    @JSONField(name="pageCount")
    public int getPageCount(){
        return pageCount;
    }

    @JSONField(name="pageCount")
    public void setPageCount(int pageCount){
        this.pageCount = pageCount;
    }

    @JSONField(name="currentPage")
    public int getCurrentPage(){
        return currentPage;
    }

    @JSONField(name="currentPage")
    public void setCurrentPage(int currentPage){
        this.currentPage = currentPage;
    }

    @JSONField(name="pageSize")
    public int getPageSize(){
        return pageSize;
    }

    @JSONField(name="pageSize")
    public void setPageSize(int pageSize){
        this.pageSize = pageSize;
    }

    @JSONField(name="list")
    public CompanyVo[] getList(){
        return list;
    }

    @JSONField(name="list")
    public void setList(CompanyVo[] list){
        this.list = list;
    }

    @JSONField(name="hasMore")
    public boolean getHasMore(){
        return hasMore;
    }

    @JSONField(name="hasMore")
    public void setHasMore(boolean hasMore){
        this.hasMore = hasMore;
    }

}

    }

    @JSONField(name="id")
    public void setId(int id){
        this.id = id;
    }

    @JSONField(name="cityId")
    public int getCityId(){
        return cityId;
    }

    @JSONField(name="cityId")
    public void setCityId(int cityId){
        this.cityId = cityId;
    }

    @JSONField(name="address")
    public String getAddress(){
        return address;
    }

    @JSONField(name="address")
    public void setAddress(String address){
        this.address = address;
    }

    @JSONField(name="latitude")
    public double getLatitude(){
        return latitude;
    }

    @JSONField(name="latitude")
    public void setLatitude(double latitude){
        this.latitude = latitude;
    }

    @JSONField(name="longitude")
    public double getLongitude(){
        return longitude;
    }

    @JSONField(name="longitude")
    public void setLongitude(double longitude){
        this.longitude = longitude;
    }

}
