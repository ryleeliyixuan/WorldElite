public class ApiResult implements Serializable{

    private int code; //0为成功，其他错误码参考 ApiCode
    private String msg; //错误消息
    private UserCorporateTagVo[] data; //数据包

    @JSONField(name="code")
    public int getCode(){
        return code;
    }

    @JSONField(name="code")
    public void setCode(int code){
        this.code = code;
    }

    @JSONField(name="msg")
    public String getMsg(){
        return msg;
    }

    @JSONField(name="msg")
    public void setMsg(String msg){
        this.msg = msg;
    }

    @JSONField(name="data")
    public UserCorporateTagVo[] getData(){
        return data;
    }

    @JSONField(name="data")
    public void setData(UserCorporateTagVo[] data){
        this.data = data;
    }

}

public class UserCorporateTagVo implements Serializable{

    private long id; //标签ID
    private long jobApplyId; //职位申请ID
    private String tagName; //企业给个人用户设置的标签
    private String createTime; //标签创建时间

    @JSONField(name="id")
    public long getId(){
        return id;
    }

    @JSONField(name="id")
    public void setId(long id){
        this.id = id;
    }

    @JSONField(name="jobApplyId")
    public long getJobApplyId(){
        return jobApplyId;
    }

    @JSONField(name="jobApplyId")
    public void setJobApplyId(long jobApplyId){
        this.jobApplyId = jobApplyId;
    }

    @JSONField(name="tagName")
    public String getTagName(){
        return tagName;
    }

    @JSONField(name="tagName")
    public void setTagName(String tagName){
        this.tagName = tagName;
    }

    @JSONField(name="createTime")
    public String getCreateTime(){
        return createTime;
    }

    @JSONField(name="createTime")
    public void setCreateTime(String createTime){
        this.createTime = createTime;
    }

}
