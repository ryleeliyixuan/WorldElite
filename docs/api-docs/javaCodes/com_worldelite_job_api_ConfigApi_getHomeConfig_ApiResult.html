public class ApiResult implements Serializable{

    private int code; //0为成功，其他错误码参考 ApiCode
    private String msg; //错误消息
    private HomeConfigForm data; //数据包

    @JSONField(name="code")
    public int getCode(){
        return code;
    }

    @JSONField(name="code")
    public void setCode(int code){
        this.code = code;
    }

    @JSONField(name="msg")
    public String getMsg(){
        return msg;
    }

    @JSONField(name="msg")
    public void setMsg(String msg){
        this.msg = msg;
    }

    @JSONField(name="data")
    public HomeConfigForm getData(){
        return data;
    }

    @JSONField(name="data")
    public void setData(HomeConfigForm data){
        this.data = data;
    }

}

public class HomeConfigForm implements Serializable{

    private int configType; //配置类型
    private String mainTitle; //主标题
    private String subTitle; //副标题
    private String picUrl; //图片

    @JSONField(name="configType")
    public int getConfigType(){
        return configType;
    }

    @JSONField(name="configType")
    public void setConfigType(int configType){
        this.configType = configType;
    }

    @JSONField(name="mainTitle")
    public String getMainTitle(){
        return mainTitle;
    }

    @JSONField(name="mainTitle")
    public void setMainTitle(String mainTitle){
        this.mainTitle = mainTitle;
    }

    @JSONField(name="subTitle")
    public String getSubTitle(){
        return subTitle;
    }

    @JSONField(name="subTitle")
    public void setSubTitle(String subTitle){
        this.subTitle = subTitle;
    }

    @JSONField(name="picUrl")
    public String getPicUrl(){
        return picUrl;
    }

    @JSONField(name="picUrl")
    public void setPicUrl(String picUrl){
        this.picUrl = picUrl;
    }

}
