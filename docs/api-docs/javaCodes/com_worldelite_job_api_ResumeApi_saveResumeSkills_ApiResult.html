public class ApiResult implements Serializable{

    private int code; //0为成功，其他错误码参考 ApiCode
    private String msg; //错误消息
    private ResumeSkillVo[] data; //数据包

    @JSONField(name="code")
    public int getCode(){
        return code;
    }

    @JSONField(name="code")
    public void setCode(int code){
        this.code = code;
    }

    @JSONField(name="msg")
    public String getMsg(){
        return msg;
    }

    @JSONField(name="msg")
    public void setMsg(String msg){
        this.msg = msg;
    }

    @JSONField(name="data")
    public ResumeSkillVo[] getData(){
        return data;
    }

    @JSONField(name="data")
    public void setData(ResumeSkillVo[] data){
        this.data = data;
    }

}

public class ResumeSkillVo implements Serializable{

    private int id; //技能ID
    private String name; //技能名
    private byte level; //等级

    @JSONField(name="id")
    public int getId(){
        return id;
    }

    @JSONField(name="id")
    public void setId(int id){
        this.id = id;
    }

    @JSONField(name="name")
    public String getName(){
        return name;
    }

    @JSONField(name="name")
    public void setName(String name){
        this.name = name;
    }

    @JSONField(name="level")
    public byte getLevel(){
        return level;
    }

    @JSONField(name="level")
    public void setLevel(byte level){
        this.level = level;
    }

}
