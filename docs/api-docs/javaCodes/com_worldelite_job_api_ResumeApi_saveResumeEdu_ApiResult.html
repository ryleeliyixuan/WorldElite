public class ApiResult implements Serializable{

    private int code; //0为成功，其他错误码参考 ApiCode
    private String msg; //错误消息
    private ResumeEduVo data; //数据包

    @JSONField(name="code")
    public int getCode(){
        return code;
    }

    @JSONField(name="code")
    public void setCode(int code){
        this.code = code;
    }

    @JSONField(name="msg")
    public String getMsg(){
        return msg;
    }

    @JSONField(name="msg")
    public void setMsg(String msg){
        this.msg = msg;
    }

    @JSONField(name="data")
    public ResumeEduVo getData(){
        return data;
    }

    @JSONField(name="data")
    public void setData(ResumeEduVo data){
        this.data = data;
    }

}

public class ResumeEduVo implements Serializable{

    private byte showEditFlag; //
    private int id; //学历ID
    private String schoolName; //学校名
    private String majorName; //专业
    private String startTime; //开始时间
    private String finishTime; //结束时间
    private DictVo degree; //学历
    private double gpa; //gpa

    @JSONField(name="showEditFlag")
    public byte getShowEditFlag(){
        return showEditFlag;
    }

    @JSONField(name="showEditFlag")
    public void setShowEditFlag(byte showEditFlag){
        this.showEditFlag = showEditFlag;
    }

    @JSONField(name="id")
    public int getId(){
        return id;
    }

    @JSONField(name="id")
    public void setId(int id){
        this.id = id;
    }

    @JSONField(name="schoolName")
    public String getSchoolName(){
        return schoolName;
    }

    @JSONField(name="schoolName")
    public void setSchoolName(String schoolName){
        this.schoolName = schoolName;
    }

    @JSONField(name="majorName")
    public String getMajorName(){
        return majorName;
    }

    @JSONField(name="majorName")
    public void setMajorName(String majorName){
        this.majorName = majorName;
    }

    @JSONField(name="startTime")
    public String getStartTime(){
        return startTime;
    }

    @JSONField(name="startTime")
    public void setStartTime(String startTime){
        this.startTime = startTime;
    }

    @JSONField(name="finishTime")
    public String getFinishTime(){
        return finishTime;
    }

    @JSONField(name="finishTime")
    public void setFinishTime(String finishTime){
        this.finishTime = finishTime;
    }

    @JSONField(name="degree")
    public DictVo getDegree(){
        return degree;
    }

    @JSONField(name="degree")
    public void setDegree(DictVo degree){
        this.degree = degree;
    }

    @JSONField(name="gpa")
    public double getGpa(){
        return gpa;
    }

    @JSONField(name="gpa")
    public void setGpa(double gpa){
        this.gpa = gpa;
    }

}

public class DictVo implements Serializable{

    private int id; //字典ID
    private String name; //字典名
    private int type; //字典类型
    private String value; //字典值

    @JSONField(name="id")
    public int getId(){
        return id;
    }

    @JSONField(name="id")
    public void setId(int id){
        this.id = id;
    }

    @JSONField(name="name")
    public String getName(){
        return name;
    }

    @JSONField(name="name")
    public void setName(String name){
        this.name = name;
    }

    @JSONField(name="type")
    public int getType(){
        return type;
    }

    @JSONField(name="type")
    public void setType(int type){
        this.type = type;
    }

    @JSONField(name="value")
    public String getValue(){
        return value;
    }

    @JSONField(name="value")
    public void setValue(String value){
        this.value = value;
    }

}
