public class ApiResult implements Serializable{

    private int code; //0为成功，其他错误码参考 ApiCode
    private String msg; //错误消息
    private ResumeExpVo data; //数据包

    @JSONField(name="code")
    public int getCode(){
        return code;
    }

    @JSONField(name="code")
    public void setCode(int code){
        this.code = code;
    }

    @JSONField(name="msg")
    public String getMsg(){
        return msg;
    }

    @JSONField(name="msg")
    public void setMsg(String msg){
        this.msg = msg;
    }

    @JSONField(name="data")
    public ResumeExpVo getData(){
        return data;
    }

    @JSONField(name="data")
    public void setData(ResumeExpVo data){
        this.data = data;
    }

}

public class ResumeExpVo implements Serializable{

    private byte showEditFlag; //
    private int id; //工作经验ID
    private String startTime; //开始时间
    private String finishTime; //结束时间
    private String company; //公司名
    private String depart; //部门
    private String post; //职位
    private String description; //描述
    private int position; //排序
    private byte onWork; //是在在职

    @JSONField(name="showEditFlag")
    public byte getShowEditFlag(){
        return showEditFlag;
    }

    @JSONField(name="showEditFlag")
    public void setShowEditFlag(byte showEditFlag){
        this.showEditFlag = showEditFlag;
    }

    @JSONField(name="id")
    public int getId(){
        return id;
    }

    @JSONField(name="id")
    public void setId(int id){
        this.id = id;
    }

    @JSONField(name="startTime")
    public String getStartTime(){
        return startTime;
    }

    @JSONField(name="startTime")
    public void setStartTime(String startTime){
        this.startTime = startTime;
    }

    @JSONField(name="finishTime")
    public String getFinishTime(){
        return finishTime;
    }

    @JSONField(name="finishTime")
    public void setFinishTime(String finishTime){
        this.finishTime = finishTime;
    }

    @JSONField(name="company")
    public String getCompany(){
        return company;
    }

    @JSONField(name="company")
    public void setCompany(String company){
        this.company = company;
    }

    @JSONField(name="depart")
    public String getDepart(){
        return depart;
    }

    @JSONField(name="depart")
    public void setDepart(String depart){
        this.depart = depart;
    }

    @JSONField(name="post")
    public String getPost(){
        return post;
    }

    @JSONField(name="post")
    public void setPost(String post){
        this.post = post;
    }

    @JSONField(name="description")
    public String getDescription(){
        return description;
    }

    @JSONField(name="description")
    public void setDescription(String description){
        this.description = description;
    }

    @JSONField(name="position")
    public int getPosition(){
        return position;
    }

    @JSONField(name="position")
    public void setPosition(int position){
        this.position = position;
    }

    @JSONField(name="onWork")
    public byte getOnWork(){
        return onWork;
    }

    @JSONField(name="onWork")
    public void setOnWork(byte onWork){
        this.onWork = onWork;
    }

}
