public class ApiResult implements Serializable{

    private int code; //0为成功，其他错误码参考 ApiCode
    private String msg; //错误消息
    private UserExpectJobVo data; //数据包

    @JSONField(name="code")
    public int getCode(){
        return code;
    }

    @JSONField(name="code")
    public void setCode(int code){
        this.code = code;
    }

    @JSONField(name="msg")
    public String getMsg(){
        return msg;
    }

    @JSONField(name="msg")
    public void setMsg(String msg){
        this.msg = msg;
    }

    @JSONField(name="data")
    public UserExpectJobVo getData(){
        return data;
    }

    @JSONField(name="data")
    public void setData(UserExpectJobVo data){
        this.data = data;
    }

}

public class UserExpectJobVo implements Serializable{

    private JobCategoryVo[] categoryList; //期望职位
    private DictVo[] cityList; //期望城市
    private int minSalary; //最低薪资
    private int maxSalary; //最高薪资

    @JSONField(name="categoryList")
    public JobCategoryVo[] getCategoryList(){
        return categoryList;
    }

    @JSONField(name="categoryList")
    public void setCategoryList(JobCategoryVo[] categoryList){
        this.categoryList = categoryList;
    }

    @JSONField(name="cityList")
    public DictVo[] getCityList(){
        return cityList;
    }

    @JSONField(name="cityList")
    public void setCityList(DictVo[] cityList){
        this.cityList = cityList;
    }

    @JSONField(name="minSalary")
    public int getMinSalary(){
        return minSalary;
    }

    @JSONField(name="minSalary")
    public void setMinSalary(int minSalary){
        this.minSalary = minSalary;
    }

    @JSONField(name="maxSalary")
    public int getMaxSalary(){
        return maxSalary;
    }

    @JSONField(name="maxSalary")
    public void setMaxSalary(int maxSalary){
        this.maxSalary = maxSalary;
    }

}

public class JobCategoryVo implements Serializable{

    private int id; //类型ID
    private String name; //类型名
    private int parentId; //父类型
    private int position; //排序

    @JSONField(name="id")
    public int getId(){
        return id;
    }

    @JSONField(name="id")
    public void setId(int id){
        this.id = id;
    }

    @JSONField(name="name")
    public String getName(){
        return name;
    }

    @JSONField(name="name")
    public void setName(String name){
        this.name = name;
    }

    @JSONField(name="parentId")
    public int getParentId(){
        return parentId;
    }

    @JSONField(name="parentId")
    public void setParentId(int parentId){
        this.parentId = parentId;
    }

    @JSONField(name="position")
    public int getPosition(){
        return position;
    }

    @JSONField(name="position")
    public void setPosition(int position){
        this.position = position;
    }

}

public class DictVo implements Serializable{

    private int id; //字典ID
    private String name; //字典名
    private int type; //字典类型
    private String value; //字典值

    @JSONField(name="id")
    public int getId(){
        return id;
    }

    @JSONField(name="id")
    public void setId(int id){
        this.id = id;
    }

    @JSONField(name="name")
    public String getName(){
        return name;
    }

    @JSONField(name="name")
    public void setName(String name){
        this.name = name;
    }

    @JSONField(name="type")
    public int getType(){
        return type;
    }

    @JSONField(name="type")
    public void setType(int type){
        this.type = type;
    }

    @JSONField(name="value")
    public String getValue(){
        return value;
    }

    @JSONField(name="value")
    public void setValue(String value){
        this.value = value;
    }

}
