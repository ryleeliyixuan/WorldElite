public class User implements Serializable{

    private String name; //名字
    private int age; //年龄

    @JSONField(name="name")
    public String getName(){
        return name;
    }

    @JSONField(name="name")
    public void setName(String name){
        this.name = name;
    }

    @JSONField(name="age")
    public int getAge(){
        return age;
    }

    @JSONField(name="age")
    public void setAge(int age){
        this.age = age;
    }

}
