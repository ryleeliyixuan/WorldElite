public class ApiResult implements Serializable{

    private int code; //0为成功，其他错误码参考 ApiCode
    private String msg; //错误消息
    private JobDescExampleVo[] data; //数据包

    @JSONField(name="code")
    public int getCode(){
        return code;
    }

    @JSONField(name="code")
    public void setCode(int code){
        this.code = code;
    }

    @JSONField(name="msg")
    public String getMsg(){
        return msg;
    }

    @JSONField(name="msg")
    public void setMsg(String msg){
        this.msg = msg;
    }

    @JSONField(name="data")
    public JobDescExampleVo[] getData(){
        return data;
    }

    @JSONField(name="data")
    public void setData(JobDescExampleVo[] data){
        this.data = data;
    }

}

public class JobDescExampleVo implements Serializable{

    private long id; //主键Id
    private long categoryId; //职位类型Id
    private String description; //职位描述
    private String createTime; //创建时间

    @JSONField(name="id")
    public long getId(){
        return id;
    }

    @JSONField(name="id")
    public void setId(long id){
        this.id = id;
    }

    @JSONField(name="categoryId")
    public long getCategoryId(){
        return categoryId;
    }

    @JSONField(name="categoryId")
    public void setCategoryId(long categoryId){
        this.categoryId = categoryId;
    }

    @JSONField(name="description")
    public String getDescription(){
        return description;
    }

    @JSONField(name="description")
    public void setDescription(String description){
        this.description = description;
    }

    @JSONField(name="createTime")
    public String getCreateTime(){
        return createTime;
    }

    @JSONField(name="createTime")
    public void setCreateTime(String createTime){
        this.createTime = createTime;
    }

}
