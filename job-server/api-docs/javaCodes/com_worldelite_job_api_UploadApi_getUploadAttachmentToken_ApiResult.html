public class ApiResult implements Serializable{

    private int code; //0为成功，其他错误码参考 ApiCode
    private String msg; //错误消息
    private UploadTokenVo data; //数据包

    @JSONField(name="code")
    public int getCode(){
        return code;
    }

    @JSONField(name="code")
    public void setCode(int code){
        this.code = code;
    }

    @JSONField(name="msg")
    public String getMsg(){
        return msg;
    }

    @JSONField(name="msg")
    public void setMsg(String msg){
        this.msg = msg;
    }

    @JSONField(name="data")
    public UploadTokenVo getData(){
        return data;
    }

    @JSONField(name="data")
    public void setData(UploadTokenVo data){
        this.data = data;
    }

}

public class UploadTokenVo implements Serializable{

    private String OSSAccessKeyId; //
    private String policy; //
    private String signature; //
    private String expire; //
    private String host; //
    private String key; //
    private String successActionStatus; //

    @JSONField(name="OSSAccessKeyId")
    public String getOSSAccessKeyId(){
        return OSSAccessKeyId;
    }

    @JSONField(name="OSSAccessKeyId")
    public void setOSSAccessKeyId(String OSSAccessKeyId){
        this.OSSAccessKeyId = OSSAccessKeyId;
    }

    @JSONField(name="policy")
    public String getPolicy(){
        return policy;
    }

    @JSONField(name="policy")
    public void setPolicy(String policy){
        this.policy = policy;
    }

    @JSONField(name="signature")
    public String getSignature(){
        return signature;
    }

    @JSONField(name="signature")
    public void setSignature(String signature){
        this.signature = signature;
    }

    @JSONField(name="expire")
    public String getExpire(){
        return expire;
    }

    @JSONField(name="expire")
    public void setExpire(String expire){
        this.expire = expire;
    }

    @JSONField(name="host")
    public String getHost(){
        return host;
    }

    @JSONField(name="host")
    public void setHost(String host){
        this.host = host;
    }

    @JSONField(name="key")
    public String getKey(){
        return key;
    }

    @JSONField(name="key")
    public void setKey(String key){
        this.key = key;
    }

    @JSONField(name="success_action_status")
    public String getSuccessActionStatus(){
        return successActionStatus;
    }

    @JSONField(name="success_action_status")
    public void setSuccessActionStatus(String successActionStatus){
        this.successActionStatus = successActionStatus;
    }

}
