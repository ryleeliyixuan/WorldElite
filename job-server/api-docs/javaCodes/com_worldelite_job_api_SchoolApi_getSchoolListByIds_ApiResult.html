public class ApiResult implements Serializable{

    private int code; //0为成功，其他错误码参考 ApiCode
    private String msg; //错误消息
    private SchoolVo[] data; //数据包

    @JSONField(name="code")
    public int getCode(){
        return code;
    }

    @JSONField(name="code")
    public void setCode(int code){
        this.code = code;
    }

    @JSONField(name="msg")
    public String getMsg(){
        return msg;
    }

    @JSONField(name="msg")
    public void setMsg(String msg){
        this.msg = msg;
    }

    @JSONField(name="data")
    public SchoolVo[] getData(){
        return data;
    }

    @JSONField(name="data")
    public void setData(SchoolVo[] data){
        this.data = data;
    }

}

public class SchoolVo implements Serializable{

    private int id; //学校ID
    private String name; //学校名
    private String siteLink; //主页
    private CountryVo country; //所在国家

    @JSONField(name="id")
    public int getId(){
        return id;
    }

    @JSONField(name="id")
    public void setId(int id){
        this.id = id;
    }

    @JSONField(name="name")
    public String getName(){
        return name;
    }

    @JSONField(name="name")
    public void setName(String name){
        this.name = name;
    }

    @JSONField(name="siteLink")
    public String getSiteLink(){
        return siteLink;
    }

    @JSONField(name="siteLink")
    public void setSiteLink(String siteLink){
        this.siteLink = siteLink;
    }

    @JSONField(name="country")
    public CountryVo getCountry(){
        return country;
    }

    @JSONField(name="country")
    public void setCountry(CountryVo country){
        this.country = country;
    }

}

public class CountryVo implements Serializable{

    private int id; //ID
    private String phoneCode; //电话区号
    private String countryCode; //国家代号
    private String chineseName; //中文名
    private String englishName; //英文名

    @JSONField(name="id")
    public int getId(){
        return id;
    }

    @JSONField(name="id")
    public void setId(int id){
        this.id = id;
    }

    @JSONField(name="phoneCode")
    public String getPhoneCode(){
        return phoneCode;
    }

    @JSONField(name="phoneCode")
    public void setPhoneCode(String phoneCode){
        this.phoneCode = phoneCode;
    }

    @JSONField(name="countryCode")
    public String getCountryCode(){
        return countryCode;
    }

    @JSONField(name="countryCode")
    public void setCountryCode(String countryCode){
        this.countryCode = countryCode;
    }

    @JSONField(name="chineseName")
    public String getChineseName(){
        return chineseName;
    }

    @JSONField(name="chineseName")
    public void setChineseName(String chineseName){
        this.chineseName = chineseName;
    }

    @JSONField(name="englishName")
    public String getEnglishName(){
        return englishName;
    }

    @JSONField(name="englishName")
    public void setEnglishName(String englishName){
        this.englishName = englishName;
    }

}
