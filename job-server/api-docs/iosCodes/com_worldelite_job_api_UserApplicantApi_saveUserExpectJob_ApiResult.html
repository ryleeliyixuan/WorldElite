#import <Foundation/Foundation>;

@interface ApiResult : NSObject

//0为成功，其他错误码参考 ApiCode
@property (nonatomic, strong) NSInteger code;
//错误消息
@property (nonatomic, assign) NSString msg;
//数据包
@property (nonatomic, strong) UserExpectJobVo data;


@end



#import <Foundation/Foundation>;

@interface UserExpectJobVo : NSObject

//期望职位
@property (nonatomic, strong) JobCategoryVo[] categoryList;
//期望城市
@property (nonatomic, strong) DictVo[] cityList;
//最低薪资
@property (nonatomic, strong) NSInteger minSalary;
//最高薪资
@property (nonatomic, strong) NSInteger maxSalary;


@end



#import <Foundation/Foundation>;

@interface JobCategoryVo : NSObject

//类型ID
@property (nonatomic, strong) NSInteger id;
//类型名
@property (nonatomic, assign) NSString name;
//父类型
@property (nonatomic, strong) NSInteger parentId;
//排序
@property (nonatomic, strong) NSInteger position;


@end



#import <Foundation/Foundation>;

@interface DictVo : NSObject

//字典ID
@property (nonatomic, strong) NSInteger id;
//字典名
@property (nonatomic, assign) NSString name;
//字典类型
@property (nonatomic, strong) NSInteger type;
//字典值
@property (nonatomic, assign) NSString value;


@end
