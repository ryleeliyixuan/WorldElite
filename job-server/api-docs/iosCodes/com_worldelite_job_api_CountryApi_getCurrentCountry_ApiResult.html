#import <Foundation/Foundation>;

@interface ApiResult : NSObject

//0为成功，其他错误码参考 ApiCode
@property (nonatomic, strong) NSInteger code;
//错误消息
@property (nonatomic, assign) NSString msg;
//数据包
@property (nonatomic, strong) CountryVo data;


@end



#import <Foundation/Foundation>;

@interface CountryVo : NSObject

//ID
@property (nonatomic, strong) NSInteger id;
//电话区号
@property (nonatomic, assign) NSString phoneCode;
//国家代号
@property (nonatomic, assign) NSString countryCode;
//中文名
@property (nonatomic, assign) NSString chineseName;
//英文名
@property (nonatomic, assign) NSString englishName;


@end
